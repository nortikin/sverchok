class ModifierNode

class ModifierNode:
    @property
    def sv_internal_links(self):
        mapping = [
            (self.inputs[0], self.outputs[0]),
            (self.inputs[1], self.outputs[1]),
            (self.inputs[2], self.outputs[2]),
        ]
        if 'FaceData' in self.outputs:
            mapping.append((self.inputs['FaceData'], self.outputs['FaceData']))
        return mapping
